 30def set_high_priority(process):

 80def parse_list(value):
 81    """
 82    Parses a string representation of a list and returns the parsed list.
 83
 84    Args:
 85        value (str): The string representation of the list.
 86
 87    Returns:
 88        list: The parsed list, which can contain integers, floats, or strings.
 89
 90    Raises:
 91        ValueError: If the input value is not a valid list format or contains mixed types or unsupported types.
 92    """
 93    try:
 94        parsed_value = ast.literal_eval(value)
 95        if isinstance(parsed_value, list):
 96            # Check if all elements are homogeneous (either all int, float, or str)
 97            if all(isinstance(item, (int, float, str)) for item in parsed_value):
 98                return parsed_value
 99            else:
100                raise ValueError("List contains mixed types or unsupported types")
101        elif isinstance(parsed_value, tuple):
102            # Convert tuple to list if it’s a single-element tuple
103            return list(parsed_value) if len(parsed_value) > 1 else [parsed_value[0]]
104        else:
105            raise ValueError(f"Expected a list but got {type(parsed_value).__name__}")
106    except (ValueError, SyntaxError) as e:
107        raise ValueError(f"Invalid format for list: {value}. Error: {e}")

631def download_dataset(q, repo_id, subfolder, local_dir=None, retries=5, delay=5):
632    """
633    Downloads a dataset or settings files from Hugging Face and returns the local path.
634
635    Args:
636        repo_id (str): The repository ID (e.g., 'einarolafsson/toxo_mito' or 'einarolafsson/spacr_settings').
637        subfolder (str): The subfolder path within the repository (e.g., 'plate1' or the settings subfolder).
638        local_dir (str): The local directory where the files will be saved. Defaults to the user's home directory.
639        retries (int): Number of retry attempts in case of failure.
640        delay (int): Delay in seconds between retries.
641
642    Returns:
643        str: The local path to the downloaded files.
644    """
645    if local_dir is None:
646        local_dir = os.path.join(os.path.expanduser("~"), "datasets")
647
648    local_subfolder_dir = os.path.join(local_dir, subfolder if subfolder else "settings")
649    if not os.path.exists(local_subfolder_dir):
650        os.makedirs(local_subfolder_dir)
651    elif len(os.listdir(local_subfolder_dir)) > 0:
652        q.put(f"Files already downloaded to: {local_subfolder_dir}")
653        return local_subfolder_dir
654
655    attempt = 0
656    while attempt < retries:
657        try:
658            files = list_repo_files(repo_id, repo_type="dataset")
659            subfolder_files = [file for file in files if file.startswith(subfolder) or (subfolder == "" and file.endswith('.csv'))]
660
661            for file_name in subfolder_files:
662                for download_attempt in range(retries):
663                    try:
664                        url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{file_name}?download=true"
665                        response = requests.get(url, stream=True)
666                        response.raise_for_status()
667
668                        local_file_path = os.path.join(local_subfolder_dir, os.path.basename(file_name))
669                        with open(local_file_path, 'wb') as file:
670                            for chunk in response.iter_content(chunk_size=8192):
671                                file.write(chunk)
672                        q.put(f"Downloaded file: {file_name}")
673                        break
674                    except (requests.HTTPError, requests.Timeout) as e:
675                        q.put(f"Error downloading {file_name}: {e}. Retrying in {delay} seconds...")
676                        time.sleep(delay)
677                else:
678                    raise Exception(f"Failed to download {file_name} after multiple attempts.")
679
680            return local_subfolder_dir
681
682        except (requests.HTTPError, requests.Timeout) as e:
683            q.put(f"Error downloading files: {e}. Retrying in {delay} seconds...")
684            attempt += 1
685            time.sleep(delay)
686
687    raise Exception("Failed to download files after multiple attempts.")

723        # If the frame is already cached at the current size, return it
724        if (frame_idx, frame_width, frame_height) in resized_frames_cache:
