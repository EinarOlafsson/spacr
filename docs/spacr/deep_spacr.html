 154def evaluate_model_performance(model, loader, epoch, loss_type):
 155    """
 156    Evaluates the performance of a model on a given data loader.
 157
 158    Args:
 159        model (torch.nn.Module): The model to evaluate.
 160        loader (torch.utils.data.DataLoader): The data loader to evaluate the model on.
 161        loader_name (str): The name of the data loader.
 162        epoch (int): The current epoch number.
 163        loss_type (str): The type of loss function to use.
 164
 165    Returns:
 166        data_df (pandas.DataFrame): The classification metrics data as a DataFrame.
 167        prediction_pos_probs (list): The positive class probabilities for each prediction.
 168        all_labels (list): The true labels for each prediction.
 169    """
 170    
 171    from .utils import calculate_loss
 172    
 173    def classification_metrics(all_labels, prediction_pos_probs):
 174        """
 175        Calculate classification metrics for binary classification.
 176
 177        Parameters:
 178        - all_labels (list): List of true labels.
 179        - prediction_pos_probs (list): List of predicted positive probabilities.
 180        - loader_name (str): Name of the data loader.
 181
 182        Returns:
 183        - data_df (DataFrame): DataFrame containing the calculated metrics.
 184        """
 185
 186        if len(all_labels) != len(prediction_pos_probs):
 187            raise ValueError(f"all_labels ({len(all_labels)}) and pred_labels ({len(prediction_pos_probs)}) have different lengths")
 188
 189        unique_labels = np.unique(all_labels)
 190        if len(unique_labels) >= 2:
 191            pr_labels = np.array(all_labels).astype(int)
 192            precision, recall, thresholds = precision_recall_curve(pr_labels, prediction_pos_probs, pos_label=1)
 193            pr_auc = auc(recall, precision)
 194            thresholds = np.append(thresholds, 0.0)
 195            f1_scores = 2 * (precision * recall) / (precision + recall)
 196            optimal_idx = np.nanargmax(f1_scores)
 197            optimal_threshold = thresholds[optimal_idx]
 198            pred_labels = [int(p > 0.5) for p in prediction_pos_probs]
 199        if len(unique_labels) < 2:
 200            optimal_threshold = 0.5
 201            pred_labels = [int(p > optimal_threshold) for p in prediction_pos_probs]
 202            pr_auc = np.nan
 203        data = {'label': all_labels, 'pred': pred_labels}
 204        df = pd.DataFrame(data)
 205        pc_df = df[df['label'] == 1.0]
 206        nc_df = df[df['label'] == 0.0]
 207        correct = df[df['label'] == df['pred']]
 208        acc_all = len(correct) / len(df)
 209        if len(pc_df) > 0:
 210            correct_pc = pc_df[pc_df['label'] == pc_df['pred']]
 211            acc_pc = len(correct_pc) / len(pc_df)
 212        else:
 213            acc_pc = np.nan
 214        if len(nc_df) > 0:
 215            correct_nc = nc_df[nc_df['label'] == nc_df['pred']]
 216            acc_nc = len(correct_nc) / len(nc_df)
 217        else:
 218            acc_nc = np.nan
 219        data_dict = {'accuracy': acc_all, 'neg_accuracy': acc_nc, 'pos_accuracy': acc_pc, 'prauc':pr_auc, 'optimal_threshold':optimal_threshold}
 220        return data_dict
 221    
 222    device = torch.device("cuda:0" if torch.cuda.is_available() else "cpu")
 223    model.eval()
 224    loss = 0
 225    correct = 0
 226    total_samples = 0
 227    prediction_pos_probs = []
 228    all_labels = []
 229    model = model.to(device)
 230    with torch.no_grad():
 231        for batch_idx, (data, target, _) in enumerate(loader, start=1):
 232            start_time = time.time()
 233            data, target = data.to(device), target.to(device).float()
 234            output = model(data)
 235            loss += F.binary_cross_entropy_with_logits(output, target, reduction='sum').item()
 236            loss = calculate_loss(output, target, loss_type=loss_type)
 237            loss += loss.item()
 238            total_samples += data.size(0)
 239            pred = torch.where(output >= 0.5,
 240                               torch.Tensor([1.0]).to(device).float(),
 241                               torch.Tensor([0.0]).to(device).float())
 242            correct += pred.eq(target.view_as(pred)).sum().item()
 243            batch_prediction_pos_prob = torch.sigmoid(output).cpu().numpy()
 244            prediction_pos_probs.extend(batch_prediction_pos_prob.tolist())
 245            all_labels.extend(target.cpu().numpy().tolist())
 246            mean_loss = loss / total_samples
 247            acc = correct / total_samples
 248            end_time = time.time()
 249            test_time = end_time - start_time
 250            #print(f'\rTest: epoch: {epoch} Accuracy: {acc:.5f} batch: {batch_idx+1}/{len(loader)} loss: {mean_loss:.5f} loss: {mean_loss:.5f} time {test_time:.5f}', end='\r', flush=True)
 251    
 252    loss /= len(loader)
 253    data_dict = classification_metrics(all_labels, prediction_pos_probs)
 254    data_dict['loss'] = loss.item()
 255    data_dict['epoch'] = epoch
 256    data_dict['Accuracy'] = acc
 257    
 258    return data_dict, [prediction_pos_probs, all_labels]

 836class SmoothGrad:
 837    def __init__(self, model, n_samples=50, stdev_spread=0.15):
 838        self.model = model
 839        self.n_samples = n_samples
 840        self.stdev_spread = stdev_spread
 841
 842    def compute_smooth_grad(self, input_tensor, target_class):
 843        self.model.eval()
 844        stdev = self.stdev_spread * (input_tensor.max() - input_tensor.min())
 845        total_gradients = torch.zeros_like(input_tensor)
 846        
 847        for i in range(self.n_samples):
 848            noise = torch.normal(mean=0, std=stdev, size=input_tensor.shape).to(input_tensor.device)
 849            noisy_input = input_tensor + noise
 850            noisy_input.requires_grad_()
 851            output = self.model(noisy_input)
 852            self.model.zero_grad()
 853            output[0, target_class].backward()
 854            total_gradients += noisy_input.grad
 855
 856        avg_gradients = total_gradients / self.n_samples
 857        return avg_gradients.abs()
