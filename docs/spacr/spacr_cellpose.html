243def compare_mask(args):
244    src, filename, dirs, conditions = args
245    paths = [os.path.join(d, filename) for d in dirs]
246
247    if not all(os.path.exists(path) for path in paths):
248        return None
249
250    from .io import _read_mask
251    from .utils import boundary_f1_score, compute_segmentation_ap, jaccard_index
252
253    masks = [_read_mask(path) for path in paths]
254    file_results = {'filename': filename}
255
256    for i in range(len(masks)):
257        for j in range(i + 1, len(masks)):
258            mask_i, mask_j = masks[i], masks[j]
259            f1_score = boundary_f1_score(mask_i, mask_j)
260            jac_index = jaccard_index(mask_i, mask_j)
261            ap_score = compute_segmentation_ap(mask_i, mask_j)
262
263            file_results.update({
264                f'jaccard_{conditions[i]}_{conditions[j]}': jac_index,
265                f'boundary_f1_{conditions[i]}_{conditions[j]}': f1_score,
266                f'ap_{conditions[i]}_{conditions[j]}': ap_score
267            })
268    
269    return file_results

271def compare_cellpose_masks(src, verbose=False, processes=None, save=True):
272    from .plot import visualize_cellpose_masks, plot_comparison_results
273    from .io import _read_mask
274
275    dirs = [os.path.join(src, d) for d in os.listdir(src) if os.path.isdir(os.path.join(src, d)) and d != 'results']
276    dirs.sort()
277    conditions = [os.path.basename(d) for d in dirs]
278
279    # Get common files in all directories
280    common_files = set(os.listdir(dirs[0]))
281    for d in dirs[1:]:
282        common_files.intersection_update(os.listdir(d))
283    common_files = list(common_files)
284
285    # Create a pool of n_jobs
286    with Pool(processes=processes) as pool:
287        args = [(src, filename, dirs, conditions) for filename in common_files]
288        results = pool.map(compare_mask, args)
289
290    # Filter out None results (from skipped files)
291    results = [res for res in results if res is not None]
292    print(results)
293    if verbose:
294        for result in results:
295            filename = result['filename']
296            masks = [_read_mask(os.path.join(d, filename)) for d in dirs]
297            visualize_cellpose_masks(masks, titles=conditions, filename=filename, save=save, src=src)
298
299    fig = plot_comparison_results(results)
300    save_results_and_figure(src, fig, results)
301    return
